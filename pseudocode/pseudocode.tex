\begin{algorithm}[H]
\SetAlgoLined
\SetKwFunction{Minimax}{Minimax}
\SetKwProg{Fn}{Function}{:}{}

% % \Input{game state $state$, depth $depth$, maximizing player $maxPlayer$}
% % \Output{best value}

\Fn{\Minimax{$state$, $depth$, $maximizing$ = \textbf{true}}}{
    \If{$depth = 0$ or game is over}{
        \Return{evaluate($state$)}
    }
    
    \If{$maximizing$ = \textbf{true}}{
        $bestValue \gets -\infty$\;
        \For{\textbf{each} child \textbf{in} $state$}{
            $value \gets$ \Minimax{child, $depth - 1$, \textbf{false}}\;
            $bestValue \gets \max(bestValue, value)$\;
        }
        \Return{$bestValue$}\;
    }
    \Else{
        $bestValue \gets \infty$\;
        \For{\textbf{each} child \textbf{in} $state$}{
            $value \gets$ \Minimax{child, $depth - 1$, \textbf{true}}\;
            $bestValue \gets \min(bestValue, value)$\;
        }
        \Return{$bestValue$}\;
    }
}

\caption{Minimax Algorithm}
\end{algorithm}

\begin{algorithm}[H]
\SetAlgoLined
\SetKwFunction{AlphaBeta}{AlphaBeta}
\SetKwProg{Fn}{Function}{:}{}

\Fn{\AlphaBeta{$state$, $depth$, $\alpha$, $\beta$, $maximizing$ = \textbf{true}}}{
    \If{$depth = 0$ or game is over}{
        \Return{evaluate($state$)}
    }
    
    \If{$maximizing$ = \textbf{true}}{
        $bestValue \gets -\infty$\;
        \For{\textbf{each} child \textbf{in} $state$}{
            $value \gets$ \AlphaBeta{child, $depth - 1$, $\alpha$, $\beta$, \textbf{false}}\;
            $bestValue \gets \max(bestValue, value)$\;
            $\alpha \gets \max(\alpha, value)$\;
            \If{$\beta \leq \alpha$}{
                break\;
            }
        }
        \Return{$bestValue$}\;
    }
    \Else{
        $bestValue \gets \infty$\;
        \For{\textbf{each} child \textbf{in} $state$}{
            $value \gets$ \AlphaBeta{child, $depth - 1$, $\alpha$, $\beta$, \textbf{true}}\;
            $bestValue \gets \min(bestValue, value)$\;
            $\beta \gets \min(\beta, value)$\;
            \If{$\beta \leq \alpha$}{
                break\;
            }
        }
        \Return{$bestValue$}\;
    }
}
\caption{Minimax Algorithm with Alpha-Beta Pruning}
\end{algorithm}
